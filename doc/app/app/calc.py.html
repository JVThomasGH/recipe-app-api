"""
Calculator functions
"""


def add(x, y):
    """Add x and y and return result."""
    return x + y


def subtract(x, y):
    """Subtract x from y and return result."""
    return y - x


def multiply(x, y):
    """Multiply x and y and return result."""
    return x * y


def divide(x, y):
    """Divide x by y and return result."""
    return x / y
